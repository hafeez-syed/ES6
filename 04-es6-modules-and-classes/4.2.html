Module sequence


                        1. File script.4.2.base.js
                        ____________

                        // log from loaded module i.e module_loading_sequence.js will be displayed
                        // first, and then the rest of the console logs, because import statements
                        // gets Hoisted.
                        console.log('starting base.js');
                        import {PI} from 'js/module_loading_sequence.js';
                        console.log('ending base.js');
                        console.log('check above the Hoisted module import log');

                        2. File script.4.2.module_loading_sequence.js
                        ____________

                        export const PI = 3.14;
                        console.log('Hello from a hoisted "module loading sequence.js" with PI value:' + PI);

                        _________________________________________________________

                        
default


                        1. File script.4.2.base.js
                        ____________

                        // notice, we have exported name as `default` in 'default.js'
                        // and, imported `myName` instead of `{name}` in 'base.js'.
                        // When exporting default we can use custom alias (with no curly braces) ]
                        import myName from 'js/default.js';
                        console.log('my name is ' + myName);

                        OR

                        // We can also use `{default as customVariable}`
                        import {default as myFullName} from 'js/default.js';
                        console.log('My Full name is ' + myFullName);


                        2. File script.4.2.default.js
                        ____________

                        export const PI = 3.14;
                        let name = 'Hafeez Syed';

                        export script.4.2.default name;

ambiguous or undefined

                        import someValue from 'js/script.4.2.ambiguous.js';

exporting default


                        1. File script.4.2.base.js
                        ____________

                        // since `default` value is provided in 'exporting_default.js'
                        // name will display `Syed` as it is exported as `default`
                        import name from 'js/exporting_default.js';
                        console.log('default name is ' + name, divId);


                        2. File script.4.2.exporting_default.js
                        ____________

                        let firstName = 'Hafeez';
                        let lastName = 'Syed';

                        export {firstName, lastName as default};
                        _________________________________________________________

                        
wildcard


                        1. File script.4.2.base.js
                        ____________

                        // wildcard `*` with alias `values` will become an object
                        import * as values from 'js/wildcard.js';
                        consoleAndDisplayResults('value is ::: ' , divId);
                        consoleAndDisplayResults(values.email, divId);
                        consoleAndDisplayResults(values.phone, divId);


                        2. File script.4.2.wildcard.js
                        ____________
