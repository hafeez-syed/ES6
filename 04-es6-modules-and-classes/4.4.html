                        1. File script.4.3.base.js
                        ____________

                        // It will output `99` since the `projectId`
                        // property of object `project` is mutable
                        import {project} from 'script.4.3.module1';
                        project.projectId = 8000;
                        console.log(project.projectId);

                        2. File script.4.3.module1
                        ____________

                        export let projectId = 99;
                        export let project = {
                            projectId: 99
                        };
                        _________________________________________________________

                        1. File script.4.3.base.js
                        ____________

                        // The value of `project2.projectId` has been updated
                        // from `8000` to `999`
                        // So when you call the `showProject()` method
                        // It will output `999` instead of `8000`
                        import {project2, showProject} from 'script.4.3.module1';
                        project2.projectId = 999;
                        showProject();
                        console.log(project2.projectId);

                        2. File script.4.3.module1
                        ____________

                        export let projectId = 99;

                        export let project = {
                            projectId: 99
                        };

                        export let project2 = {
                            projectId: 8000
                        };

                        export showProject() {
                            console.log(project2.projectId);
                        }

                        import {showOriginal, updateFunction} from 'js/script.4.3.module1.js';

                        2. File script.4.3.module1
                        ____________

                        export let projectId = 99;

                        export let project = {
                            projectId: 99
                        };

                        export let project2 = {
                            projectId: 8000
                        };

                        export showProject() {
                            console.log(project2.projectId);
                        }

                        export function showOriginal(divId) {
                            console.log('in original');
                        }

                        export function updateFunction(divId) {
                            showOriginal = function () {
                                console.log('in updated');
                            };
                        }
