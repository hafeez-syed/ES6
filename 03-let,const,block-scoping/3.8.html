03.08. Octal and Binary Literals

                    
                        // Octal
                        var value = 0o10;

                        console.log(value);

                        _________________________________________________________

                        // Octal
                        var value = 0O10;

                        console.log(value);

                        _________________________________________________________

                        // Binary
                        var value = 0b10;

                        console.log(value);

                        _________________________________________________________

                        // Binary
                        var value = 0B10;

                        console.log(value);

                        _________________________________________________________
