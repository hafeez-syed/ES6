03.11. Advanced Destructing

                    
                        let [high, low] = [,];

                        console.log(`high: ${high} low: ${low}`);

                        _________________________________________________________

                        // Destructuring requires an iterator so undefined is not allowed
                        let [high, low] = undefined;

                        console.log(`high: ${high} low: ${low}`);

                        _________________________________________________________

                        // Destructuring requires an iterator so null is not allowed
                        let [high, low] = null;

                        console.log(`high: ${high} low: ${low}`);

                        _________________________________________________________

                        try {
                            let [high, low] = undefined;
                        }
                        catch(e) {
                            console.log(e);
                        }

                        _________________________________________________________

                        let [high, low, ] = [500, 200];

                        console.log(`high: ${high} low: ${low}`);

                        _________________________________________________________

                        for(let [a, b] of [[5, 10]]) {
                            console.log(`${a} ${b}`);
                        }

                        _________________________________________________________

                        let count = 0;
                        for(let [a, b] of [[5, 10]]) {
                            console.log(`${a} ${b}`);
                            count++;
                        }
                        console.log(count);
