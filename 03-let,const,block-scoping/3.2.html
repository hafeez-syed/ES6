03.02.f. block scoping - "var" v "let"

            

                
                    
                        var updateFunctions = [];
                        for (var i=0; i < 2; i++) {
                            updateFunctions.push(function(){ return i;});
                        }

                        console.log(updateFunctions[0]());

                        _________________________________________________________

                        let updateFunctions = [];
                        for (let i=0; i < 2; i++) {
                            updateFunctions.push(function(){ return i;});
                        }

                        console.log(updateFunctions[0]());
