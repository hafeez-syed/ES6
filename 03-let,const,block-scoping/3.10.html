03.10b. Destructuring Objects

            

                
                    
                        let salaryObject = {
                            lowSal: 32000,
                            averageSal: 50000,
                            highSal: 75000
                        }
                        let {lowSal, averageSal, highSal} = salaryObject;
                        console.log(highSal);

                        _________________________________________________________

                        let salaryObject1 = {
                            lowSal1: 32000,
                            averageSal1: 50000,
                            highSal1: 75000
                        }
                        let {lowSal1: newSal1, averageSal1: newAverageSal1, highSal1: newHigh1} = salaryObject1;
                        console.log(newHigh1); // We are actually renaming `highSal1` to `newHigh1` so, it will not output `undefined`

                        _________________________________________________________

                        let salaryObject2 = {
                            lowSal2: 32000,
                            averageSal2: 50000,
                            highSal2: 75000
                        }
                        let newSal2, newAverageSal2, newHigh2;
                        {lowSal2: newSal2, averageSal2: newAverageSal2, highSal2: newHigh2} = salaryObject2;
                        console.log(newHigh2); // Since we have not used object literals ie, `{}`, it will output `Syntax Error`

                        _________________________________________________________

                        let salaryObject3 = {
                            lowSal3: 32000,
                            averageSal3: 50000,
                            highSal3: 75000
                        }
                        let newSal3, newAverageSal3, newHigh3;
                        ({lowSal3: newSal3, averageSal3: newAverageSal3, highSal3: newHigh3} = salaryObject3);
                        console.log(newHigh3); // We have now rapped the above object in the parenthesis to fix the Syntax Error
