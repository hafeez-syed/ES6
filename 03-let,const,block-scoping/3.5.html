03.05 - Rest and Spread

03.05a. Rest

                    
                        // The `...` symbol is the `Rest` operator
                        // it will gather up all the remaining parameters in the function
                        var showCategories = function(productId, ...categories) {
                            console.log(categories instanceof Array);
                        };

                        showCategories(123, 'search', 'advertising');

                        _________________________________________________________
                        var showCategories = function(productId, ...categories) {
                            console.log(categories);
                        };

                        showCategories(123, 'search', 'advertising');

                        _________________________________________________________
                        var showCategories = function(productId, ...categories) {
                            console.log(categories);
                        };

                        showCategories(123);

                        _________________________________________________________
                        var showCategories = function(productId, ...categories) {
                        };

                        showCategories(showCategories.length);

                        _________________________________________________________
                        var showCategories = function(productId, ...categories) {
                            console.log(arguments.length);
                        };

                        showCategories(123, 'search', 'advertising');

                        _________________________________________________________
                        var showCategories = new Function('...categories', 'return categories');

                        console.log(showCategories('search', 'advertising'));

                        _________________________________________________________
                    
                

            


            
03.05b. Spread

            

                
                    
                        // The `...` symbol is the `Spread` operator

                        var prices = [12, 20, 18];
                        var maxPrices = Math.max(...prices);
                        console.log(maxPrices);

                        _________________________________________________________
                        var prices = [12, 20, 18];
                        var newPriceArray = [...prices];
                        console.log(newPriceArray);

                        _________________________________________________________
                        var newPriceArray = Array(...[,,]);
                        showCategories(newPriceArray);

                        _________________________________________________________
                        var newPriceArray = [...[,,]];
                        showCategories(newPriceArray);

                        _________________________________________________________
                        var maxCode = Math.max(...'43210');
                        console.log(maxCode);

                        _________________________________________________________
                        var codeArray = ['A', ...'BCD', 'E'];
                        console.log(codeArray);

                        _________________________________________________________
